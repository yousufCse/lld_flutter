```dart
import 'package:dartz/dartz.dart';
import 'package:niramoy_health_app/core/app/logger/app_logger.dart';
import 'package:niramoy_health_app/core/errors/errors.dart';
import 'package:niramoy_health_app/core/network/network_info.dart';
import 'package:niramoy_health_app/core/types/type_defs.dart';

abstract class BaseRepository {
  final NetworkInfo networkInfo;
  final AppLogger? logger;
  const BaseRepository({required this.networkInfo, this.logger});

  Result<T> handleException<T>(
    Future<T> Function() operation, {
    String? operationName,
  }) async {
    if (!await networkInfo.isConnected) {
      return const Left(NetworkFailure());
    }
    try {
      final result = await operation();
      return Right(result);
    } on NetworkException catch (e, s) {
      return Left(NetworkFailure(title: e.title, details: e.details));
    } on AuthException catch (e, s) {
      return Left(AuthFailure(title: e.title, details: e.details));
    } on ServerException catch (e, s) {
      return Left(ServerFailure(title: e.title, details: e.details));
    } on ApiException catch (e, s) {
      return Left(ApiFailure(statusCode: e.statusCode, title: e.title, details: e.details, code: e.code, additionalInfo: e.additionalInfo));
    } on ParsingException catch (e, s) {
      return Left(ParsingFailure(title: e.title, details: e.details));
    } on CacheException catch (e, s) {
      return Left(CacheFailure(title: e.title, details: e.details));
    } catch (e, s) {
      return Left(CustomFailure(title: 'Unknown Error', details: e.toString()));
    }
  }
}
```

```dart
import 'package:json_annotation/json_annotation.dart';

part 'api_response.g.dart';

@JsonSerializable(genericArgumentFactories: true, createToJson: false)
class ApiResponse<T> {
  final bool success;
  final int status;
  final String message;
  final T? data;

  const ApiResponse({required this.success, required this.status, required this.message, this.data});

  factory ApiResponse.fromJson(Map<String, dynamic> json, T Function(Object?) fromJsonT) =>
      _$ApiResponseFromJson(json, fromJsonT);

  static ApiResponse<T> fromJsonWithParser<T>(
    Map<String, dynamic> json,
    T Function(Map<String, dynamic>) fromJsonModel,
  ) {
    return ApiResponse<T>(
      success: json['success'] as bool? ?? false,
      status: json['status'] as int? ?? 0,
      message: json['message'] as String? ?? '',
      data: json['data'] != null ? fromJsonModel(json['data'] as Map<String, dynamic>) : null,
    );
  }

  ApiResponse<R> parse<R>(R Function(Map<String, dynamic>) fromJson) {
    if (data == null) throw Exception('Response data is null');
    return ApiResponse<R>(
      success: success,
      status: status,
      message: message,
      data: fromJson(data as Map<String, dynamic>),
    );
  }

  ApiResponse<List<R>> parseList<R>(R Function(Map<String, dynamic>) fromJson) {
    if (data == null) throw Exception('Response data is null');
    final list = (data as List).cast<Map<String, dynamic>>();
    return ApiResponse<List<R>>(
      success: success,
      status: status,
      message: message,
      data: list.map(fromJson).toList(),
    );
  }
}
```

**`lib/core/network/api_client.dart`** — full implementation (copy from reference):

```dart
import 'package:dio/dio.dart';
import 'package:niramoy_health_app/core/app/logger/console_app_logger.dart';
import 'package:niramoy_health_app/core/constants/app_errors.dart';
import 'package:niramoy_health_app/core/errors/errors.dart';
import 'package:niramoy_health_app/core/network/api_error_body.dart';
import 'package:niramoy_health_app/core/network/api_response.dart';

class ApiClient {
  final Dio _dio;
  ApiClient({required Dio dio}) : _dio = dio;

  Future<ApiResponse> get({required String endpoint, Map<String, dynamic>? queryParameters, Options? options}) =>
      _execute(() async => _parseResponse((await _dio.get(endpoint, queryParameters: queryParameters, options: options)).data));

  Future<ApiResponse> post({required String endpoint, dynamic data, Map<String, dynamic>? queryParameters, Options? options}) =>
      _execute(() async => _parseResponse((await _dio.post(endpoint, data: data, queryParameters: queryParameters, options: options)).data));

  Future<ApiResponse> put({required String endpoint, dynamic data, Map<String, dynamic>? queryParameters, Options? options}) =>
      _execute(() async => _parseResponse((await _dio.put(endpoint, data: data, queryParameters: queryParameters, options: options)).data));

  Future<ApiResponse> patch({required String endpoint, dynamic data, Map<String, dynamic>? queryParameters, Options? options}) =>
      _execute(() async => _parseResponse((await _dio.patch(endpoint, data: data, queryParameters: queryParameters, options: options)).data));

  Future<ApiResponse> delete({required String endpoint, dynamic data, Map<String, dynamic>? queryParameters, Options? options}) =>
      _execute(() async => _parseResponse((await _dio.delete(endpoint, data: data, queryParameters: queryParameters, options: options)).data));

  Future<void> deleteWithoutResponse({required String endpoint, dynamic data, Map<String, dynamic>? queryParameters, Options? options}) =>
      _execute(() async { await _dio.delete(endpoint, data: data, queryParameters: queryParameters, options: options); });

  Future<T> _execute<T>(Future<T> Function() operation) async {
    try {
      return await operation();
    } on DioException catch (e) {
      throw _handleDioError(e);
    } on AppException {
      rethrow;
    } catch (e, stackTrace) {
      throw CustomException(title: AppErrors.unknownError, details: AppErrors.unknownErrorDetails, originalError: e, stackTrace: stackTrace);
    }
  }

  ApiResponse _parseResponse(dynamic data) {
    if (data is Map<String, dynamic> && data.containsKey('success') && data.containsKey('data')) {
      return ApiResponse.fromJsonWithParser(data, (d) => d);
    }
    return ApiResponse(success: true, status: 200, message: '', data: data);
  }

  Never _handleDioError(DioException error) {
    logger.e('DioException: ${error.message}', error, error.stackTrace);
    switch (error.type) {
      case DioExceptionType.connectionTimeout:
      case DioExceptionType.sendTimeout:
      case DioExceptionType.receiveTimeout:
        throw NetworkException(title: AppErrors.timeout, details: AppErrors.timeoutDetails, originalError: error, stackTrace: error.stackTrace);
      case DioExceptionType.badResponse:
        throw _handleResponseError(error.response, error: error);
      case DioExceptionType.connectionError:
        throw NetworkException(title: AppErrors.noInternet, details: AppErrors.noInternetDetails, originalError: error, stackTrace: error.stackTrace);
      case DioExceptionType.badCertificate:
        throw ServerException(title: AppErrors.badCertificate, details: AppErrors.badCertificateDetails, statusCode: -1, originalError: error, stackTrace: error.stackTrace);
      case DioExceptionType.cancel:
        throw CustomException(title: AppErrors.requestCancelled, details: AppErrors.requestCancelledDetails, originalError: error, stackTrace: error.stackTrace);
      case DioExceptionType.unknown:
        if (error.message?.contains('SocketException') ?? false) {
          throw NetworkException(title: AppErrors.noInternet, details: AppErrors.noInternetDetails, originalError: error, stackTrace: error.stackTrace);
        }
        throw ServerException(title: AppErrors.unknownError, details: AppErrors.unknownErrorDetails, statusCode: -1, originalError: error, stackTrace: error.stackTrace);
    }
  }

  AppException _handleResponseError(Response? response, {required DioException error}) {
    if (response == null) return ServerException(title: AppErrors.serverError, details: AppErrors.serverErrorDetails, statusCode: -1, originalError: error, stackTrace: error.stackTrace);
    final Map<String, dynamic>? body = response.data is Map<String, dynamic> ? response.data : null;
    final errorBody = ApiErrorBody.fromField(body?['error']);
    final legacyMessage = body?['message'] as String?;
    final title = errorBody.title;
    final detail = errorBody.detail ?? legacyMessage;
    switch (response.statusCode ?? -1) {
      case 401: return AuthException.unauthorized();
      case 403: return AuthException(title: title ?? AppErrors.unauthorized, details: detail ?? AppErrors.unauthorizedDetails, originalError: error, stackTrace: error.stackTrace);
      case >= 500: return ServerException(title: title ?? AppErrors.serverError, details: detail ?? AppErrors.serverErrorDetails, statusCode: response.statusCode!, originalError: error, stackTrace: error.stackTrace);
      default: return ApiException(title: title ?? AppErrors.apiRequestFailed, details: detail ?? AppErrors.apiRequestFailed, statusCode: response.statusCode!, code: errorBody.code, additionalInfo: errorBody.additionalInfo, originalError: error, stackTrace: error.stackTrace);
    }
  }
}
```

```dart

import 'package:niramoy_health_app/core/constants/app_errors.dart';
import 'package:niramoy_health_app/core/errors/exceptions/data_exceptions.dart';
import 'package:shared_preferences/shared_preferences.dart';

abstract class LocalCache {
  Future<void> put(String key, String value);
  Future<String?> get(String key, {Duration? ttl});
  Future<void> remove(String key);
  Future<void> clear();
}

@LazySingleton(as: LocalCache)
class LocalCacheImpl implements LocalCache {
  final SharedPreferencesAsync _prefs;
  LocalCacheImpl(this._prefs);

  String _dataKey(String key) => 'cache:$key:data';
  String _tsKey(String key) => 'cache:$key:ts';

  @override
  Future<void> put(String key, String value) async {
    try {
      await _prefs.setString(_dataKey(key), value);
      await _prefs.setString(_tsKey(key), DateTime.now().millisecondsSinceEpoch.toString());
    } catch (e, s) { throw CacheException(title: AppErrors.cacheError, details: AppErrors.cacheWriteError, originalError: e, stackTrace: s); }
  }

  @override
  Future<String?> get(String key, {Duration? ttl}) async {
    try {
      final data = await _prefs.getString(_dataKey(key));
      if (data == null) return null;
      if (ttl != null) {
        final tsRaw = await _prefs.getString(_tsKey(key));
        if (tsRaw == null) return null;
        if (DateTime.now().difference(DateTime.fromMillisecondsSinceEpoch(int.parse(tsRaw))) > ttl) return null;
      }
      return data;
    } catch (e, s) { throw CacheException(title: AppErrors.cacheError, details: AppErrors.cacheReadError, originalError: e, stackTrace: s); }
  }

  @override
  Future<void> remove(String key) async {
    try { await _prefs.remove(_dataKey(key)); await _prefs.remove(_tsKey(key)); }
    catch (e, s) { throw CacheException(title: AppErrors.cacheError, details: AppErrors.cacheOperationFailed, originalError: e, stackTrace: s); }
  }

  @override
  Future<void> clear() async {
    try { await _prefs.clear(); }
    catch (e, s) { throw CacheException(title: AppErrors.cacheError, details: AppErrors.cacheOperationFailed, originalError: e, stackTrace: s); }
  }
}
```

## ARCHITECTURE RULES TO ENFORCE AT ALL TIMES

1. **Domain layer** never imports from `data/` or `presentation/`
2. **Repository methods** always return `Result<T>` wrapped in `handleException()` — never throw
3. **No `print()`** — use `logger.d/i/w/e()` from `console_app_logger.dart`
4. **Cubit states** always use Freezed — never Equatable
5. **Feature cubits** `@injectable` — provided per-screen via `BlocProvider` in route
6. **Global cubits** `@lazySingleton` — registered in `global_app_providers.dart`
7. **Request models**: `@JsonSerializable(createFactory: false)` + `fromParams()` + `toJson()`
8. **Response models**: `@JsonSerializable(createToJson: false)` + `fromJson()` + `toEntity` getter
9. **Always run codegen** after adding/changing any annotated class
10. **Always use `fvm flutter`** / `fvm dart` — never bare `flutter` / `dart`
